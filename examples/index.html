MithrilJS + Bootstrap Modal

Usage example

var modal = new Modal();

var app = {
    view: function () {
        return m('div', [
            m('.btn.btn-primary', {onclick: modal.show.bind(modal)}, 'Show modal'),
            modal.view({
                header: function () {
                    return m('h4.modal-title', 'modal title goes here');
                },
                body: function () {
                    return m('p', 'modal body goes here');
                },
                footer: function () {
                    return m('a.btn.btn-default', {
                        onclick: modal.hide.bind(modal)
                    }, 'Close');
                }
            })
        ]);
    }
};

m.mount(document.getElementById('view'), app);
